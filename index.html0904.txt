<script>
// ============================================================
// ⚠️  REMPLACEZ PAR VOTRE URL GOOGLE APPS SCRIPT
// ============================================================
var SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbzzU_2dZkW2bwBeBCnIvZ-2kP1o0x2WSp_qtVrMuLNavau8T2ePyeVGMPaPlvC1JQ0ajQ/exec';
// ============================================================

var demandeurs = [], responsables = [], themes = [];

// Date minimum = aujourd'hui
document.getElementById('inp-date').min = new Date().toISOString().split('T')[0];

// ---- API helpers ----
function apiGet(action) {
  return fetch(SCRIPT_URL + '?action=' + action, {redirect:'follow'})
    .then(function(r){ return r.json(); });
}
function apiPost(data) {
  return fetch(SCRIPT_URL, {method:'POST', body:JSON.stringify(data), redirect:'follow'})
    .then(function(r){ return r.json(); });
}

// ---- Chargement initial ----
function charger() {
  document.getElementById('form-wrap').innerHTML = '<div class="loading-msg">⏳ Chargement...</div>';

  Promise.all([apiGet('demandeurs'), apiGet('responsables'), apiGet('themes')])
    .then(function(results) {
      demandeurs   = results[0] || [];
      responsables = results[1] || [];
      themes       = results[2] || [];
      buildForm();
    })
    .catch(function() {
      document.getElementById('form-wrap').innerHTML =
        '<div class="loading-msg" style="color:#DC2626">❌ Erreur de connexion. Vérifiez la configuration de l\'URL.</div>';
    });
}

function buildForm() {
  // Reconstruire le HTML du formulaire
  document.getElementById('form-wrap').innerHTML = getFormHTML();

  // Remettre date min
  document.getElementById('inp-date').min = new Date().toISOString().split('T')[0];

  // Remplir demandeurs
  var selD = document.getElementById('sel-dem');
  demandeurs.forEach(function(d) {
    var o = document.createElement('option');
    o.value = d.ID;
    o.textContent = d.Nom + ' ' + d.Prenom + ' — ' + d.Matricule;
    selD.appendChild(o);
  });

  // Remplir responsables
  var selR = document.getElementById('sel-resp');
  responsables.forEach(function(r) {
    var o = document.createElement('option');
    o.value = r.Nom + ' ' + r.Prenom;
    o.textContent = r.Nom + ' ' + r.Prenom;
    selR.appendChild(o);
  });

  // Remplir thèmes
  var selT = document.getElementById('sel-theme');
  themes.forEach(function(t) {
    var o = document.createElement('option');
    o.value = t.Titre;
    o.disabled = t.Complet;
    o.textContent = t.Titre + (t.Complet
      ? ' — COMPLET'
      : ' (' + t.Restantes + ' place' + (t.Restantes > 1 ? 's' : '') + ')');
    selT.appendChild(o);
  });
}

function getFormHTML() {
  return '<div class="card">'
    + '<div class="card-title"><span class="num">1</span> Identification du demandeur</div>'
    + '<div class="field"><label>Sélectionner votre nom <span class="req">*</span></label>'
    + '<select id="sel-dem" onchange="changerDemandeur()"><option value="">-- Choisir un demandeur --</option></select></div>'
    + '<div class="info-grid" id="info-grid" style="display:none">'
    + '<div class="info-cell"><small>Nom</small><span id="i-nom"></span></div>'
    + '<div class="info-cell"><small>Prénom</small><span id="i-prenom"></span></div>'
    + '<div class="info-cell"><small>Matricule</small><span id="i-mat"></span></div>'
    + '<div class="info-cell"><small>Poste</small><span id="i-poste"></span></div>'
    + '</div></div>'
    + '<div class="card">'
    + '<div class="card-title"><span class="num">2</span> Responsable</div>'
    + '<div class="field"><label>Choisir votre responsable <span class="req">*</span></label>'
    + '<select id="sel-resp"><option value="">-- Choisir un responsable --</option></select></div></div>'
    + '<div class="card">'
    + '<div class="card-title"><span class="num">3</span> Formation souhaitée</div>'
    + '<div class="field"><label>Thème de formation <span class="req">*</span></label>'
    + '<select id="sel-theme" onchange="changerTheme()"><option value="">-- Choisir un thème --</option></select>'
    + '<div class="prog-wrap" id="prog-wrap" style="display:none">'
    + '<div class="prog-bar"><div class="prog-fill" id="prog-fill"></div></div>'
    + '<span class="prog-label" id="prog-label"></span></div></div>'
    + '<div class="field"><label>Date proposée <span class="req">*</span></label>'
    + '<input type="date" id="inp-date" min=""></div></div>'
    + '<button class="btn-submit" id="btn-submit" onclick="soumettre()">Soumettre la demande →</button>';
}

// ---- Events ----
function changerDemandeur() {
  var id = document.getElementById('sel-dem').value;
  var d  = demandeurs.find(function(x){ return String(x.ID) === id; });
  var grid = document.getElementById('info-grid');
  if (d) {
    document.getElementById('i-nom').textContent    = d.Nom;
    document.getElementById('i-prenom').textContent = d.Prenom;
    document.getElementById('i-mat').textContent    = d.Matricule;
    document.getElementById('i-poste').textContent  = d.Poste;
    grid.style.display = 'grid';
  } else {
    grid.style.display = 'none';
  }
}

function changerTheme() {
  var titre = document.getElementById('sel-theme').value;
  var t = themes.find(function(x){ return x.Titre === titre; });
  var wrap = document.getElementById('prog-wrap');
  cacherAlerte();
  if (t) {
    var pct = Math.min((t.Occupees / t.MaxPlaces) * 100, 100);
    document.getElementById('prog-fill').style.width = pct + '%';
    document.getElementById('prog-fill').style.background = t.Complet ? '#EF4444' : '#1B4FBF';
    document.getElementById('prog-label').textContent = t.Occupees + '/' + t.MaxPlaces;
    wrap.style.display = 'flex';
    if (t.Complet) afficherAlerte('Les places pour "' + t.Titre + '" sont épuisées. Contactez l\'administration.');
  } else {
    wrap.style.display = 'none';
  }
}

function soumettre() {
  var demId = document.getElementById('sel-dem').value;
  var resp  = document.getElementById('sel-resp').value;
  var theme = document.getElementById('sel-theme').value;
  var date  = document.getElementById('inp-date').value;

  if (!demId) { alert('Veuillez sélectionner un demandeur.'); return; }
  if (!resp)  { alert('Veuillez sélectionner un responsable.'); return; }
  if (!theme) { alert('Veuillez sélectionner un thème.'); return; }
  if (!date)  { alert('Veuillez saisir une date.'); return; }

  var d = demandeurs.find(function(x){ return String(x.ID) === demId; });
  var t = themes.find(function(x){ return x.Titre === theme; });
  if (t && t.Complet) { afficherAlerte('Les places pour cette formation sont épuisées.'); return; }

  var btn = document.getElementById('btn-submit');
  btn.disabled = true;
  btn.innerHTML = '<span class="spin"></span> Envoi en cours...';
  cacherAlerte();

  apiPost({
    action:'submit',
    nom:d.Nom, prenom:d.Prenom, mat:d.Matricule, poste:d.Poste,
    resp:resp, theme:theme, date:date, p:''
  }).then(function(res) {
    if (res.err) {
      afficherAlerte(res.err);
    } else {
      document.getElementById('form-wrap').style.display = 'none';
      document.getElementById('success-card').className = 'success-card show';
    }
    btn.disabled = false;
    btn.innerHTML = 'Soumettre la demande →';
  }).catch(function() {
    afficherAlerte('Erreur de connexion. Réessayez.');
    btn.disabled = false;
    btn.innerHTML = 'Soumettre la demande →';
  });
}

function afficherAlerte(msg) {
  document.getElementById('alert-msg').textContent = msg;
  document.getElementById('alert-box').className = 'alert-box show';
  window.scrollTo(0, 0);
}
function cacherAlerte() {
  document.getElementById('alert-box').className = 'alert-box';
}

function resetForm() {
  document.getElementById('success-card').className = 'success-card';
  document.getElementById('form-wrap').style.display = 'block';
  cacherAlerte();
  charger();
}

// Démarrage
charger();
</script>
